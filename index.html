    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const generateIdeasBtn = document.getElementById('generate-ideas-btn');

            const communityChallengeInput = document.getElementById('community-challenge-input');

            const actionIdeasOutput = document.getElementById('action-ideas-output');

            const loadingIndicator = document.getElementById('loading-indicator');


            generateIdeasBtn.addEventListener('click', async () => {

                const prompt = communityChallengeInput.value.trim();


                if (!prompt) {

                    actionIdeasOutput.innerHTML = '<p class="text-red-400">Please enter a community challenge or goal.</p>';

                    return;

                }


                loadingIndicator.classList.remove('hidden');

                actionIdeasOutput.innerHTML = ''; // Clear previous output


                try {

                    let chatHistory = [];

                    chatHistory.push({ role: "user", parts: [{ text: `Given the E.P.I.C. movement's focus on empowerment, resilience, cultural pride, family values, self-love, education, and entrepreneurship, suggest concrete, actionable steps to address the following community challenge: ${prompt}. Provide the response as a bulleted list.` }] });


                    const payload = { contents: chatHistory };

                    // IMPORTANT: The Gemini API Key is automatically provided by the platform

                    // when running this in a Canvas environment.

                    // If you are deploying this externally, you will need to replace this

                    // empty string with your actual Gemini API key.

                    // For production, use server-side calls or environment variables

                    // to keep your API key secure.

                    const apiKey = ""; // Keep this empty; Canvas will provide it at runtime.

                    const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    const result = await response.json();


                    if (result.candidates && result.candidates.length > 0 &&

                        result.candidates[0].content && result.candidates[0].content.parts &&

                        result.candidates[0].content.parts.length > 0) {

                        const text = result.candidates[0].content.parts[0].text;

                        actionIdeasOutput.innerHTML = `<p class="text-gray-200 whitespace-pre-wrap">${text}</p>`;

                    } else {

                        actionIdeasOutput.innerHTML = '<p class="text-red-400">Sorry, I could not generate ideas at this time. Please try again.</p>';

                        console.error("Unexpected API response structure:", result);

                    }

                } catch (error) {

                    actionIdeasOutput.innerHTML = '<p class="text-red-400">An error occurred while fetching ideas. Please check your network connection.</p>';

                    console.error("Error calling Gemini API:", error);

                } finally {

                    loadingIndicator.classList.add('hidden');

                }

            });

        });

    </script>
